\newcommand{\ChapFrame}[3]{%
    \AddEverypageHook{%
        \ifodd\value{page}
            \backgroundsetup{contents={%
            \begin{tikzpicture}[overlay,remember picture]
              \node[
                fill=#2,
                inner sep=0pt,
                rectangle,
                text width=2cm,
                text height=30cm,
                align=center,
                anchor=north east
              ] 
              at ($ (current page.north east) + (-0cm,-0*\thechapshift cm) $) 
              {\rotatebox{270}{\hspace*{-26cm}\parbox[c][1cm][t]{8cm}{%
                  \raggedright\textcolor{black}{\huge #1}}}};
            \end{tikzpicture}%
          }%
        }
        \else
            \backgroundsetup{contents={%
            \begin{tikzpicture}[overlay,remember picture]
            \node[
              fill=#2,
              inner sep=0pt,
              rectangle,
              text width=2cm,
              text height=30cm,
              align=center,
              anchor=north west
            ] 
            at ($ (current page.north west) + (-0cm,-0*\thechapshift cm) $) 
            {\rotatebox{90}{\hspace*{#3}\parbox[c][1cm][t]{8cm}{%
                \raggedright\textcolor{black}{\huge #1}}}};
            \end{tikzpicture}%
          }%
        }
        \fi
        \BgMaterial
    }%
  %\stepcounter{chapshift}%
}
